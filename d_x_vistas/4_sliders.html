  <h1>Sliders</h1>
  <cmp-sld>
    <label>
      Muestra resultados
      <input type="range" min="1"
          max="10">
    </label>
  </cmp-sld>
  <script>
    class CmpSld
      extends HTMLElement {
      connectedCallback() {
        setTimeout(
          () => this.prepara(),
          100);
      }
      prepara() {
        /** @private */
        this._input =
          this.querySelector(
            "input");
        if (this._input) {
          this._input.
            addEventListener(
              "input", () =>
              this.analiza());
          this.analiza();
        }
      }
      analiza() {
        const i = this._input;
        if (i) {
          const v =
            i.valueAsNumber;
          const min =
            parseFloat(i.min);
          const max =
            parseFloat(i.max);
          const value =
            (v - min) /
            (max - min) * 100;
          i.style.background = `linear-gradient(to right, var(--colSld) 0%, var(--colSld) ${value}%, var(--colSldBk) ${value}%, var(--colSldBk) 100%)`;
        }
      }
    }
    customElements.define(
      "cmp-sld", CmpSld);
  </script>
