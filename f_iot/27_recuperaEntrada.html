const Client =
  require("@replit/database");
const b = require('../b');

/**
 * @param {import("express").
 *        Request} req
 * @param {import("express").
 *        Response} resp
 */
async function
  recuperaEntrada(req, resp) {
  // @ts-ignore
  const bd = new Client(b);
  const id = req.params.id;
  /** @type {import("./tipos").
   *        Registro} */
  // @ts-ignore
  const entrada =
    await bd.get(`entrada${id}`);
  if (entrada) {
    resp.send(
      entrada.valor.toString());
  } else {
    resp.send("");
  }
}

module.exports = recuperaEntrada;
