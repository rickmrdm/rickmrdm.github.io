import {
  URL_ENTRADA
} from "./util.js";

/**
 * @param {string} dispostivo
 * @param {number} valor */
export async function
  proxyAgregaEntrada(dispostivo,
    valor) {
  let error = "";
  try {
    const res = await fetch(
      URL_ENTRADA + dispostivo,
      {
        method: "POST",
        body: valor.toString()
      });
    if (!res.ok) {
      error = res.statusText;
    }
  } catch (e) {
    error = e.message;
  }
  return error;
}
