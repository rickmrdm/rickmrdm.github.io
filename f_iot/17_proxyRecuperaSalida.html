import {
  ResInt
} from "./ResInt.js";
import {
  URL_SALIDA
} from "./util.js";

/** @param {string} dispositivo */
export async function
  proxyRecuperaSalida(
    dispositivo) {
  let error = "";
  let valor = 0;
  try {
    const res = await fetch(
      URL_SALIDA + dispositivo);
    if (res.ok) {
      const recuperado = parseInt(
        await res.text());
      if (!isNaN(recuperado)) {
        valor = recuperado;
      }
    } else {
      error = res.statusText;
    }
  } catch (e) {
    error = e.message;
  }
  return new ResInt(
    valor, error);
}
