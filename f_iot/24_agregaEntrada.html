lib/util.js

const Client =
  require("@replit/database");
const uuidAPIKey =
  require('uuid-apikey');
const b = require('../b');

/**
 * @param {import("express").
 *        Request} req
 * @param {import("express").
 *        Response} res
 */
async function
  agregaEntrada(req, res) {
  // @ts-ignore
  const bd = new Client(b);
  const id = req.params.id;
  const valor =
    parseInt(req.body, 10);
  if (!isNaN(valor)) {
    /**
     * @type {import("./tipos").
     *        Registro} */
    const entrada = {
      tipo: "Entrada",
      valor
    };
    await bd.set(`entrada${id}`,
      entrada);
    const uuid =
      // @ts-ignore
      uuidAPIKey.create();
    /**
     * @type {import("./tipos").
     *               Registro} */
    const histórico = {
      tipo: "Histórico",
      dispositivoId: id,
      valor,
      timestamp: Date.now()
    };
    await bd.set(uuid.uuid,
      histórico);
  }
  res.send("");
}

module.exports = agregaEntrada;
