import {
  URL_SALIDA
} from "./util.js";

/**
 * @param {string} dispostivo
 * @param {number} valor */
export async function
  proxyAgregaSalida(dispostivo,
    valor) {
  const res = await fetch(
    URL_SALIDA + dispostivo,
    {
      method: "POST",
      body: valor.toString()
    });
  if (!res.ok) {
    throw new Error(
      res.statusText);
  }
}
