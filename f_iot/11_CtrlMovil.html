import "./mi-footer.js";
import {
  proxyAgregaSalida
} from "./proxyAgregaSalida.js";
import {
  proxyRecuperaEntrada
} from "./proxyRecuperaEntrada.js";
import {
  proxyRecuperaSalida
} from "./proxyRecuperaSalida.js";

/** Milisegundos para realizar
 * otro loop. */
const RETRASO = 1000;

/**
 * @type {HTMLOutputElement|null}
 */
const ctDispositivo = document.
  querySelector("#ctDispositivo");
/**
 * @type {HTMLInputElement|null}
 */
const ctSalida = document.
  querySelector("#ctSalida");
/**
 * @type {HTMLMeterElement|null}
 */
const ctEntrada = document.
  querySelector("#ctEntrada");

const dispositivoId = prompt(
  "Identificador de Dispositivo:",
  "1") || "";

if (ctSalida && ctDispositivo &&
  dispositivoId) {
  ctSalida.addEventListener(
    "click", modificaSalida);
  ctDispositivo.value =
    dispositivoId;
  programa();
}

async function setup() {
  try {
    await leeSalida();
    await leeEntrada();
  } catch (e) {
    muestraError(e);
  }
}

async function loop() {
  try {
    await leeEntrada();
  } catch (e) {
    muestraError(e);
  }
}

async function leeSalida() {
  const res =
    await proxyRecuperaSalida(
      dispositivoId);
  if (res.error) {
    throw new Error(res.error);
  } else if (ctSalida) {
    ctSalida.valueAsNumber =
      res.valor;
  }
}

async function modificaSalida() {
  try {
    const valor = ctSalida ?
      ctSalida.valueAsNumber : 0;
    await proxyAgregaSalida(
      dispositivoId, valor);
  } catch (e) {
    muestraError(e);
  }
}

async function leeEntrada() {
  if (ctEntrada) {
    ctEntrada.value =
      await proxyRecuperaEntrada(
        dispositivoId);
  }
}

async function programa() {
  await setup();
  await repiteLoop();
}

async function repiteLoop() {
  await loop();
  setTimeout(repiteLoop, RETRASO);
}

/** Muestra un error en la consola
 * y un cuadro de alerta con el
 * mensaje.
 * @param {Error} e descripción
 *  del error. */
export function muestraError(e) {
  console.error(e);
  alert(e.message);
}
