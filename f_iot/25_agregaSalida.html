const { clienteMongo } =
  require("./utilsrv");

/**
 * @param {import("express").
 *        Request} req
 * @param {import("express").
 *        Response} res */
async function
  agregaSalida(req, res) {
  const client = clienteMongo();
  try {
    await client.connect();
    const iot = client.db("iot");
    const dispositivo = iot.
      collection("dispositivo");
    const id =
      parseInt(req.params.id, 10);
    const valor =
      parseInt(req.body, 10);
    if (!isNaN(valor)) {
      await dispositivo.updateOne(
        { _id: id },
        {
          $set: {
            salida: valor
          }
        });
    }
  } catch (e) {
    console.log(e);
  } finally {
    res.send("");
    client.close();
  }
}

module.exports = agregaSalida;
