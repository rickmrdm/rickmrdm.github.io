import "./mi-footer.js";
import {
  proxyAgregaEntrada
} from "./proxyAgregaEntrada.js";
import {
  proxyRecuperaSalida
} from "./proxyRecuperaSalida.js";

/** Milisegundos para realizar
 * otro loop. */
const RETRASO = 1000;

/**
 * @type {HTMLOutputElement|null}
 */
const ctDispositivo = document.
  querySelector("#ctDispositivo");
/**
 * @type {HTMLOutputElement|null}
 */
const ctSalida = document.
  querySelector("#ctSalida");
/**
 * @type {HTMLOutputElement|null}
 */
const iotError = document.
  querySelector("#iotError");
/**
 * @type {HTMLInputElement|null}
 */
const ctEntrada = document.
  querySelector("#ctEntrada");

const dispositivoId = prompt(
  "Identificador del Dispositivo",
  "1") || "";

let entrada = 0;

if (ctDispositivo &&
  dispositivoId) {
  ctDispositivo.value =
    dispositivoId;
  programa();
}

/** @param {number} valor */
function muestraSalida(valor) {
  if (ctSalida) {
    ctSalida.value =
      valor > 0 ? "ðŸ”´" : "âšª";
  }
}

function recuperaEntrada() {
  return ctEntrada ?
    ctEntrada.valueAsNumber : 0;
}

/** @param {string} mensaje */
function muestraError(mensaje) {
  console.error(mensaje);
  if (iotError) {
    iotError.value = mensaje;
  }
}

  await envÃ­aLaEntrada(true);
}

async function loop() {
  await
    muestraLaSalidaDelServidor();
  await envÃ­aLaEntrada(false);
}

async function
  muestraLaSalidaDelServidor() {
  const res =
    await proxyRecuperaSalida(
      dispositivoId);
  if (res.error.length > 0) {
    muestraError(res.error);
  } else {
    muestraSalida(res.valor);
  }
}

/** @param {boolean} forzosa */
async function
  envÃ­aLaEntrada(forzosa) {
  const nuevaEntrada =
    recuperaEntrada();
  if (forzosa ||
    entrada != nuevaEntrada) {
    let error =
      await proxyAgregaEntrada(
        dispositivoId,
        nuevaEntrada);
    if (error.length > 0) {
      muestraError(error);
      return;
    }
    entrada = nuevaEntrada;
  }
}

async function programa() {
  await setup();
  await repiteLoop();
}

async function repiteLoop() {
  await loop();
  setTimeout(repiteLoop, RETRASO);
}
