import {
  URL_ENTRADA
} from "./util.js";

/** @param {string} dispositivo */
export async function
  proxyRecuperaEntrada(
    dispositivo) {
  const res = await fetch(
    URL_ENTRADA + dispositivo);
  if (res.ok) {
    const recuperado = parseInt(
      await res.text());
    return isNaN(recuperado) ?
      0 : recuperado;
  } else {
    throw new Error(
      res.statusText);
  }
}
