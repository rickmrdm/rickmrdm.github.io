const Client =
  require("@replit/database");
const b = require('../b');

const INICIO = /* html */
  `<!DOCTYPE html>
  <html lang="es">
  <head>
    <meta charset="UTF-8">
    <title>Histórico</title>
  </head>
  <body>
    <table>
      <thead>
        <tr>
        <th>Id</th>
        <th>Dispositivo</th>
        <th>Valor</th>
        <th>Timestamp</th>
      </tr>
      </thead>
      <tbody>`;

const FINAL = /* html */
  `   </tbody>
    </table>
  </body>
  </html>`;

/**
 * @param {import("express").
 *        Request} _req
 * @param {import("express").
 *        Response} resp
 */
async function
  historico(_req, resp) {
  // @ts-ignore
  const bd = new Client(b);
  // recupera todos los registros.
  const todo = await bd.getAll();
  /**
   * @type {import("./tipos").
   *        Histórico[]} */
  const arr = [];
  for (const id in todo) {
    /** @type {import("./tipos").
     *        Registro} */
    const r = todo[id];
    if (r.tipo === "Histórico") {
      /**
       * @type {import("./tipos").
       *        Histórico} */
      const histórico = {
        id,
        dispositivoId:
          r.dispositivoId || "0",
        valor: r.valor,
        timestamp:
          r.timestamp || 0
      }
      arr.push(histórico)
    }
  }
  arr.sort((a, b) =>
    a.timestamp - b.timestamp);
  let html = INICIO;
  for (const h of arr) {
    html += /* html */
      `<tr>
        <td>${h.id}</td>
        <td>
          ${h.dispositivoId}
        </td>
        <th>${h.valor}</th>
        <th>${h.timestamp}</th>
      </tr>`;
  }
  html += FINAL;
  resp.charset = "UTF-8";
  resp.type("html").send(html);
}

module.exports = historico;
