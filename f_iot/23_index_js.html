lib/fabrica.js

const express =
  require("express");
const http = require("http");
const path = require("path");
const historico = require(
  "./servicios/historico");
const agregaEntrada = require(
  "./servicios/agregaEntrada");
const recuperaEntrada = require(
  "./servicios/recuperaEntrada");
const recuperaSalida = require(
  "./servicios/recuperaSalida");
const agregaSalida = require(
  "./servicios/agregaSalida");

const app = express();
const httpserver =
  // @ts-ignore
  http.Server(app);
const archivos = path.
  join(__dirname, "www");

app.use(express.static(archivos));
app.use(express.text());
app.get("/historico", historico);
/* La última parte del recurso se
 * marca con :i para que se tome
 * como el parámetro id. */
app.get("/entrada/:id",
  recuperaEntrada);
app.post("/entrada/:id",
  agregaEntrada);
app.get("/salida/:id",
  recuperaSalida);
app.post("/salida/:id",
  agregaSalida);
httpserver.listen(3000);
