#include "util.hpp"
#include "ResInt.hpp"
#include "proxyAgregaEntrada.hpp"
#include "proxyRecuperaSalida.hpp"

const char* const SSID = "Iot";
const char* const PASS = "12345678";
const char* const HUELLA_DIGITAL = "b2d651e280be8b90f6ce6febb7d0e66fc4cf9b6c";
const char* const DISPOSITIVO_ID = "2";

// Pin del bot√≥n flash.
const int BOTON_FLASH = 0;
// Pin de un LED.
const int LED_INTEGRADO_2 = 2;

int entrada = 0;

void muestraSalida(int valor) {
  digitalWrite(
    LED_INTEGRADO_2,
    valor > 0 ? LOW : HIGH);
}
int recuperaEntrada() {
  const int estadoDelBoton =
    digitalRead(BOTON_FLASH);
  return estadoDelBoton == HIGH ?
         10 : 2;
}
void
muestraError(String mensaje) {
  Serial.println(mensaje.c_str());
}

void setup() {
  Serial.begin(115200);
  // Serial.setDebugOutput(true);
  Serial.println();
  Serial.println();
  Serial.println();
  pinMode(
    LED_INTEGRADO_2, OUTPUT);
  pinMode(BOTON_FLASH, INPUT);
  conectaWiFi(SSID, PASS);
  muestraLaSalidaDelServidor();
  enviaLaEntrada(true);
}

void loop() {
  muestraLaSalidaDelServidor();
  enviaLaEntrada(false);
}

void
muestraLaSalidaDelServidor() {
  const ResInt res =
    proxyRecuperaSalida(
      HUELLA_DIGITAL,
      DISPOSITIVO_ID);
  if (res.error.length() > 0) {
    muestraError(res.error);
  } else {
    muestraSalida(res.valor);
  }
}

void
enviaLaEntrada(bool forzosa) {
  const int nuevaEntrada =
    recuperaEntrada();
  if (forzosa ||
      entrada != nuevaEntrada) {
    String error =
      proxyAgregaEntrada(
        HUELLA_DIGITAL,
        DISPOSITIVO_ID,
        nuevaEntrada);
    if (error.length() > 0) {
      muestraError(error);
      return;
    }
    entrada = nuevaEntrada;
  }
}
