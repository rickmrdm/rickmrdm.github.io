const crypto = require("crypto");
const Client =
  require("@replit/database");
const b = require('../b');

/**
 * @param {import("express").
 *   Request} req solicitud que
 *  recibe el  servidor web.
 * @param {import("express").
 *  Response} res respuesta que
 *  devuelve el servidor web. */
async function
  validaUsuario(req, res) {
  /* Crea un cliente que se
   * conecta a la base de
   * datos. */
  // @ts-ignore
  const bd = new Client(b);
  /* Los datos enviados por la
   * página vienen en la propiedad
   * body de req. */
  const datos = req.body;
  /* Busca en la base de datos el
   * valor asociado con el campo
   * cue en la base de datos. Si
   * no lo encuentra, recibe
   * null. */
  const pass =
    await bd.get(datos.cue);
  if (pass) {
    const enc = crypto
      .createHash("sha256")
      .update(datos.match)
      .digest("base64");
    if (pass === enc) {
      /* Devuelve a la página el
       * texto "ok". */
      res.send("ok");
      return;
    }
  }
  /* Devuelve a la página el texto
   * "". */
  res.send("");
}

module.exports = validaUsuario;
