lib/util.js

const crypto = require("crypto");
const Client =
  require("@replit/database");
const b = require('../b');

async function creaUsuarios() {
  /* Crea un cliente que se
   * conecta a la base de
   * datos. */
  // @ts-ignore
  const bd = new Client(b);
  // Encripta el texto "cuentos".
  const pass = crypto
    .createHash("sha256")
    .update("cuentos")
    .digest("base64");
  /* Agrega a la base de datos el
   * par con la llave "pepito" y
   * como valor asociado, el
   * contenido de pass. */
  await bd.set("pepito", pass);
}

module.exports = creaUsuarios;
