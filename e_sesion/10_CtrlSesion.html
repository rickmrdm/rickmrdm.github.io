const CUE = "cue";
const section = document.
  querySelector("section");
/**@type{HTMLButtonElement|null}*/
const terminar = document.
  querySelector("#terminar");
/**@type {HTMLFormElement|null}*/
const form = document["forma"];

let humanoAcreditado = false;

if (terminar) {
  terminar.addEventListener(
    "click", terminaSesión);
}
if (form) {
  form.addEventListener("submit",
    iniciaSesión);
  form.humano.addEventListener("input",
    soyHumano);
}

muestraSesion();

function soyHumano() {
  if (form) {
    humanoAcreditado =
      form.humano.
        valueAsNumber > 0;
  }
}

function muestraSesion() {
  const cue = sessionStorage.
    getItem(CUE);
  if (section) {
    section.hidden = !cue;
  }
  if (form) {
    form.hidden = Boolean(cue);
  }
}

/** @param {Event} evt */
async function iniciaSesión(evt) {
  try {
    evt.preventDefault();
    if (!form) {
      throw new Error(
        "Forma nula.");
    }
    if (!humanoAcreditado) {
      alert("Comprueba que " +
        "eres un humano.");
      return;
    }
    /* Usa el método POST para
     * enviar los datos de la
     * forma al recurso
     * "/verifica" en el
     * servidor. */
    const res = await fetch(
      "/valida",
      {
        method: "POST",
        body: new FormData(form)
      });
    if (res.ok) {
      const ver =
        await res.text();
      if (ver) {
        sessionStorage.
          setItem(CUE, ver);
        muestraSesion();
        form.cue.value = "";
        form.match.value = "";
        form.humano.
          valueAsNumber = 0;
        humanoAcreditado = false;
      } else {
        alert(
          "Datos incorrectos");
      }
    } else {
      throw new Error(
        res.statusText);
    }
  } catch (e) {
    muestraError(e);
  }
}

async function terminaSesión() {
  sessionStorage.
    removeItem(CUE);
  muestraSesion();
}

function muestraError(e) {
  console.error(e);
  alert(e.message);
}
